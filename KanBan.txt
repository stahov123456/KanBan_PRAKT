<script>
// ================================================
// Основные функции
// ================================================

const columns = {
  todo: document.getElementById('todo'),
  inprogress: document.getElementById('inprogress'),
  done: document.getElementById('done')
};

const counters = {
  todo: document.getElementById('todo-count'),
  inprogress: document.getElementById('inprogress-count'),
  done: document.getElementById('done-count')
};

let draggedCard = null;

// Обновление счётчиков
function updateCounters() {
  Object.keys(columns).forEach(key => {
    counters[key].textContent = columns[key].querySelectorAll('.card').length;
  });
}

// Создание новой карточки
function createCard(text = "Новая задача", desc = "", columnId = "todo") {
  const card = document.createElement('div');
  card.className = 'card';
  card.draggable = true;

  card.innerHTML = `
    <div class="card-title">${text}</div>
    ${desc ? `<div class="card-desc">${desc}</div>` : ''}
  `;

  // Двойной клик — редактирование
  card.addEventListener('dblclick', () => {
    const title = prompt("Название задачи:", card.querySelector('.card-title').textContent);
    if (title !== null) {
      card.querySelector('.card-title').textContent = title;
    }
    const description = prompt("Описание (можно оставить пустым):", card.querySelector('.card-desc')?.textContent || "");
    if (description !== null) {
      let descEl = card.querySelector('.card-desc');
      if (description.trim()) {
        if (!descEl) {
          descEl = document.createElement('div');
          descEl.className = 'card-desc';
          card.appendChild(descEl);
        }
        descEl.textContent = description;
      } else if (descEl) {
        descEl.remove();
      }
    }
  });

  // События перетаскивания
  card.addEventListener('dragstart', (e) => {
    draggedCard = card;
    setTimeout(() => card.style.opacity = '0.4', 0);
  });

  card.addEventListener('dragend', () => {
    setTimeout(() => {
      card.style.opacity = '1';
      draggedCard = null;
    }, 0);
  });

  // Добавляем карточку в нужную колонку
  columns[columnId].appendChild(card);
  updateCounters();
  return card;
}

// Добавление новой задачи в колонку
function addNewCardToColumn(columnId) {
  const form = document.createElement('div');
  form.className = 'form-add';
  form.innerHTML = `
    <textarea placeholder="Что нужно сделать?" autofocus></textarea>
    <button>Добавить</button>
    <button class="cancel">Отмена</button>
  `;

  const textarea = form.querySelector('textarea');
  const addBtn = form.querySelector('button');
  const cancelBtn = form.querySelector('.cancel');

  addBtn.onclick = () => {
    const text = textarea.value.trim();
    if (text) {
      createCard(text, "", columnId);
    }
    form.remove();
  };

  cancelBtn.onclick = () => form.remove();
  textarea.onkeydown = e => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      addBtn.click();
    }
    if (e.key === 'Escape') form.remove();
  };

  columns[columnId].appendChild(form);
  textarea.focus();
}

// Инициализация — добавляем кнопки "Добавить карточку"
Object.keys(columns).forEach(key => {
  const addBtn = document.createElement('div');
  addBtn.className = 'add-card';
  addBtn.textContent = '+ Добавить карточку';
  addBtn.onclick = () => addNewCardToColumn(key);
  columns[key].appendChild(addBtn);

  // Поддержка drag & drop для колонок
  columns[key].addEventListener('dragover', e => {
    e.preventDefault();
    const afterElement = getDragAfterElement(columns[key], e.clientY);
    const dragging = document.querySelector('.card[style*="opacity: 0.4"]') || draggedCard;
    if (afterElement == null) {
      columns[key].insertBefore(dragging, columns[key].querySelector('.add-card'));
    } else {
      columns[key].insertBefore(dragging, afterElement);
    }
  });

  columns[key].addEventListener('drop', e => {
    e.preventDefault();
    updateCounters();
  });
});

// Вспомогательная функция — куда вставить карточку при drop
function getDragAfterElement(column, y) {
  const draggableElements = [...column.querySelectorAll('.card:not([style*="opacity"])')];

  return draggableElements.reduce((closest, child) => {
    const box = child.getBoundingClientRect();
    const offset = y - box.top - box.height / 2;
    if (offset < 0 && offset > closest.offset) {
      return { offset: offset, element: child };
    } else {
      return closest;
    }
  }, { offset: Number.NEGATIVE_INFINITY }).element;
}
